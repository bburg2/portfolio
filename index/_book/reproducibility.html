gg_color_hue <- function(n) {
  hues = seq(15, 375, length = n + 1)
  hcl(h = hues, l = 65, c = 100)[1:n]
}

get_max_daily <- function(x){
  x <- x + c(1:4, 9:12, 17:20, 25:27)
  x <- x[x < ymd("2020-04-13 UTC")]
  return(x)
}
